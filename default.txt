server {
    # Azure App Service requires listening on 8080
    listen 8080;
    listen [::]:8080;

    server_name _;
    root /home/site/wwwroot;
    index index.php index.html;

    # 1. Main Site Logic
    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }

    # 2. Real-Time Socket Bridge (Node.js)
    # This routes /socket.io/ requests to your PM2-managed Node process
    location /socket.io/ {
        proxy_pass http://127.0.0.1:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    # 3. PHP Processing (Using TCP Port 9000 for Azure compatibility)
    location ~ \.php$ {
        fastcgi_split_path_info ^(.+\.php)(/.+)$;
        fastcgi_pass 127.0.0.1:9000;
        fastcgi_index index.php;
        include fastcgi_params;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        fastcgi_param PATH_INFO $fastcgi_path_info;
    }

    # 4. Security: Block hidden files like .htaccess
    location ~ /\.ht {
        deny all;
    }
}
