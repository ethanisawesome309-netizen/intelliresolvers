server {
    listen 8080;
    listen [::]:8080;
    root /home/site/wwwroot;
    index index.php index.html index.htm; # Put index.php first
    server_name intelliresolvers.com;
    port_in_redirect off;

    location / {
        # FIX: Try the PHP index if the file isn't found, not just html
        try_files $uri $uri/ /index.php?$query_string;
    }

    location /socket.io/ {
        proxy_pass http://127.0.0.1:3001;
        proxy_http_version 1.1;
        
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_cache_bypass $http_upgrade;
        
        # FIX: Azure requires buffering off for WebSockets to stay alive
        proxy_buffering off;
        proxy_read_timeout 86400;
    }

    location ~ \.php$ {
        # FIX: Ensure we catch .php files in subdirectories like /api/
        try_files $uri =404;

        fastcgi_split_path_info ^(.+?\.php)(/.*)$;
        fastcgi_pass 127.0.0.1:9000; 

        include fastcgi_params;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        fastcgi_param PATH_INFO $fastcgi_path_info;
    }
}
